Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="amb"(* Change to your UID *)

(* Create IPConnection and device object *)
ipcon=NETNew["Tinkerforge.IPConnection"]
c12=NETNew["Tinkerforge.BrickletCurrent12",uid,ipcon]
ipcon@Connect[host,port]

(* Set Period for current callback to 1s (1000ms) *)
(* Note: The current callback is only called every second if the *)
(*       current has changed since the last call! *)
c12@SetCurrentCallbackPeriod[1000]

(* Callback function for current callback (parameter has unit mA) *)
CurrentCB[sender_,current_]:=
 Print["Current: "<>ToString[N[Quantity[current,"mA"]]]]
AddEventHandler[c12@Current,CurrentCB]

Input["Click OK to exit"]

(* Clean up *)
ipcon@Disconnect[]
ReleaseNETObject[c12]
ReleaseNETObject[ipcon]
